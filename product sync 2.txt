import "dotenv/config";
import { envConfiguration } from "../config/env.config";

envConfiguration();

import crypto from "crypto";

import { prisma } from "../config/database.config";
import { adminDb } from "../lib/admin-db";

function computeHash(data: unknown) {
    return crypto
        .createHash("sha256")
        .update(JSON.stringify(data))
        .digest("hex");
}

const BATCH_SIZE = Number(process.env.BATCH_SIZE || 50);

async function main() {
    console.log("====================================");
    console.log("Starting Product Sync");
    console.log("====================================");

    const result = await adminDb.query(`
        SELECT
            p.id,
            p.title,
            p.slug,
            p.subtitle,
            p."shortDesc",
            p.description,
            p."pricingModel",
            p."basePrice",
            p."discountPercent",
            p."discountAmount",
            p."finalPrice",
            p."introVideoId",
            p."detailedVideoId",
            p."demoUrl",
            p."downloadUrl",
            p."trialAvailable",
            p.status,
            p."isTopProduct",
            p."isLatest",
            p."seoTitle",
            p."seoDescription",
            p.metadata,
            p."updatedAt",

            COALESCE(
                ARRAY_AGG(DISTINCT c.slug)
                FILTER (WHERE c.slug IS NOT NULL),
                '{}'
            ) as "categorySlugs",

            COALESCE(
                ARRAY_AGG(DISTINCT i.slug)
                FILTER (WHERE i.slug IS NOT NULL),
                '{}'
            ) as "industrySlugs",

            COALESCE(
                ARRAY_AGG(DISTINCT t.slug)
                FILTER (WHERE t.slug IS NOT NULL),
                '{}'
            ) as "tagSlugs"

        FROM "Product" p

        LEFT JOIN "ProductCategory" pc
            ON pc."productId" = p.id

        LEFT JOIN "Category" c
            ON c.id = pc."categoryId"

        LEFT JOIN "ProductIndustry" pi
            ON pi."productId" = p.id

        LEFT JOIN "Industry" i
            ON i.id = pi."industryId"

        LEFT JOIN "ProductTag" pt
            ON pt."productId" = p.id

        LEFT JOIN "Tag" t
            ON t.id = pt."tagId"

        GROUP BY p.id

        ORDER BY p.id ASC
    `);

    const products = result.rows;

    console.log(`Found ${products.length} products`);

    let syncedCount = 0;
    let skippedCount = 0;
    let failedCount = 0;

    const failedProducts: any[] = [];

    for (let i = 0; i < products.length; i += BATCH_SIZE) {
        const batch = products.slice(i, i + BATCH_SIZE);

        console.log(
            `\nProcessing batch ${Math.floor(i / BATCH_SIZE) + 1}`
        );

        for (const product of batch) {
            const startedAt = Date.now();

            try {
                const syncPayload = {
                    title: product.title,
                    slug: product.slug,

                    subtitle: product.subtitle,
                    shortDesc: product.shortDesc,
                    description: product.description,

                    pricingModel:
                        (product.pricingModel as any) ||
                        "ONE_TIME",

                    basePrice: product.basePrice,
                    discountPercent:
                        product.discountPercent,
                    discountAmount:
                        product.discountAmount,
                    finalPrice: product.finalPrice,

                    introVideoId:
                        product.introVideoId,

                    detailedVideoId:
                        product.detailedVideoId,

                    demoUrl: product.demoUrl,
                    downloadUrl:
                        product.downloadUrl,

                    trialAvailable:
                        product.trialAvailable,

                    status:
                        (product.status as any) ||
                        "DRAFT",

                    isTopProduct:
                        product.isTopProduct,

                    isLatest:
                        product.isLatest,

                    seoTitle:
                        product.seoTitle,

                    seoDescription:
                        product.seoDescription,

                    metadata:
                        product.metadata,

                    categorySlugs:
                        product.categorySlugs || [],

                    industrySlugs:
                        product.industrySlugs || [],

                    tagSlugs:
                        product.tagSlugs || [],
                };

                const hash =
                    computeHash(syncPayload);

                /**
                 * CHECK DUPLICATE
                 * Skip if already synced
                 */
                const existing =
                    await prisma.productCatalog.findFirst({
                        where: {
                            OR: [
                                {
                                    adminProductId:
                                        product.id,
                                },
                                {
                                    slug: product.slug,
                                },
                            ],
                        },

                        select: {
                            id: true,
                            slug: true,
                            adminProductId: true,
                            syncHash: true,
                        },
                    });

                /**
                 * SAME PRODUCT ALREADY EXISTS
                 */
                if (existing) {
                    /**
                     * Same hash = already synced
                     */
                    if (
                        existing.syncHash === hash
                    ) {
                        skippedCount++;

                        console.log(
                            `⏭ Skipped: ${product.title}`
                        );

                        continue;
                    }

                    /**
                     * Duplicate slug but different admin product
                     */
                    if (
                        existing.adminProductId !==
                        product.id
                    ) {
                        skippedCount++;

                        console.log(
                            `⚠ Duplicate slug skipped: ${product.slug}`
                        );

                        continue;
                    }

                    /**
                     * Existing product updated
                     */
                    await prisma.productCatalog.update({
                        where: {
                            id: existing.id,
                        },

                        data: {
                            ...syncPayload,

                            syncHash: hash,

                            syncStatus:
                                "SYNCED",

                            syncedAt:
                                new Date(),

                            sourceUpdatedAt:
                                product.updatedAt,

                            lastSyncAttempt:
                                new Date(),

                            syncError: null,

                            syncVersion: {
                                increment: 1,
                            },
                        },
                    });

                    console.log(
                        `🔄 Updated: ${product.title}`
                    );
                } else {
                    /**
                     * CREATE NEW
                     */
                    await prisma.productCatalog.create({
                        data: {
                            adminProductId:
                                product.id,

                            ...syncPayload,

                            syncHash: hash,

                            syncStatus:
                                "SYNCED",

                            syncedAt:
                                new Date(),

                            sourceUpdatedAt:
                                product.updatedAt,

                            lastSyncAttempt:
                                new Date(),
                        },
                    });

                    console.log(
                        `✅ Created: ${product.title}`
                    );
                }

                syncedCount++;

                await prisma.productCatalogSyncLog.create({
                    data: {
                        adminProductId:
                            product.id,

                        action: "SYNC",

                        syncStatus:
                            "SYNCED",

                        durationMs:
                            Date.now() -
                            startedAt,
                    },
                });
            } catch (error: any) {
                failedCount++;

                console.error(
                    `❌ Failed syncing ${product.title}`
                );

                console.error(error?.message);

                failedProducts.push({
                    id: product.id,
                    title: product.title,
                    error:
                        error?.message ||
                        "Unknown Error",
                });

                try {
                    await prisma.productCatalogSyncLog.create({
                        data: {
                            adminProductId:
                                product.id,

                            action: "SYNC",

                            syncStatus:
                                "ERROR",

                            error:
                                error?.message ||
                                "Unknown Error",

                            durationMs:
                                Date.now() -
                                startedAt,
                        },
                    });
                } catch { }
            }
        }

        console.log(
            `Batch completed → Synced: ${syncedCount}, Skipped: ${skippedCount}, Failed: ${failedCount}`
        );
    }

    console.log("\n====================================");
    console.log("Product Sync Completed");
    console.log("====================================");

    console.log(`Total Products: ${products.length}`);
    console.log(`Synced: ${syncedCount}`);
    console.log(`Skipped: ${skippedCount}`);
    console.log(`Failed: ${failedCount}`);

    if (failedProducts.length > 0) {
        console.log("\nFailed Products:");

        for (const item of failedProducts) {
            console.log(
                `${item.id} | ${item.title}`
            );

            console.log(item.error);

            console.log("--------------------------------");
        }
    }
}

main()
    .catch((error) => {
        console.error("Fatal Error:", error);
    })
    .finally(async () => {
        await adminDb.end();
    });
